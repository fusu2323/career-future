**完整匹配测试用例**:
```python
student_profile = {
    "name": "Zhang San",
    "education": [{"degree": "硕士", "end_date": "2025-06"}],
    "target_city": "北京",
    "mastered_skills": ["Java", "Python", "MySQL", "Spring Boot"],
    "dimensions": {
        "soft": {"details": {"communication": 80, "teamwork": 80, ...}},
        "potential": {"details": {"learning_ability": 85, ...}}
    }
}

job_profile = {
    "name": "Java Backend Developer",
    "category": "后端开发",
    "requirements": {"min_degree": "本科", "city": "北京"},
    "required_skills": ["Java", "Spring Boot", "MySQL"]
}

result = calculate_match(student_profile, job_profile)
# 结果包含:
# - total_score: 综合得分
# - dimensions: {base, skill, soft, potential}
# - weights_used: 使用的权重配置
```
